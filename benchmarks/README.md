**LU Decomposition**
```c++
for (i: 0 -> n) {
    U(i, i) = 1
}
for (i: 0 -> n) {
    for (j : i -> n) {
        L(j, i) = A(j, i);
        for (k = 0; k < i; k++) {
            L(j, i) = L(j, i) - L(j, k) * U(k, i);
        }
    }
    for (j: i + 1 -> n) {
        U(i, j) = A(i, j) / L(i, i);
        for (k = 0; k < i; k++) {
            U(i, j) = U(i, j) - ((L(i, k) * U(k, j)) / L(i, i));
        }
    }
}
```
**Cholesky Decomposition**
```c++
for (j: 0 -> N) {
    d = .0
    for (k: 0 -> j) {
        s = .0
        for (i: 0 -> k) {
            s += L(k, i) * L(j, i)
        }
        s = (A(j, k) - s) / L(k, k)
        L(j, k) = s
        d = d + s*s
    }
    d = A(j, j) - d
    L(j, j) = sqrt(max(0, d))
    for (k: j + 1 -> n) {
        L(j, k) = 0
    }
}
```
